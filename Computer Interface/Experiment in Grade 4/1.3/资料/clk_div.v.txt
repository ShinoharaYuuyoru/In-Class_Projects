module clk_div
(
	input clk,		// 100Mhz
	output clk_pulse	// 1Hz
);

reg clk_pulse = 0;
reg [25:0] div_counter = 0;

always @(posedge clk) begin
	if(div_counter >= 50000000)
		begin
			clk_pulse <= 1;
			div_counter <= 0;
		end
	else
		begin
			clk_pulse <= 0;
			div_counter <= div_counter + 1;
		end
end

endmodule
